module.exports = function towelSort(matrix) {
   if (!matrix) {
      return [];
   }
   return matrix.reduce(function (result, current, index) {
      if (index % 2) {
         return result.concat(current.sort((a, b) => b - a));
      } else {
         return result.concat(current.sort((a, b) => a - b));
      }
   }, []);
}
